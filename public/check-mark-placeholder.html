Check Mark GIF Resources

How to Add a Check Mark Animation

Your SweetAlert notifications are configured to use a check-mark animation for success messages.

Steps to add the animation:

  1. Download a check mark animation from one of the resources below
  2. Rename the file to check-mark.gif
  3. Place it in the public directory of your project

Resources for Free Check Mark GIFs:

  • GIPHY - Check Mark Search
  • LottieFiles - Check Mark (export as GIF)
  • Icons8 - Animated Check Marks
Note: You can also create your own animated check mark using tools like Adobe After Effects or online animation makers.

Alternative: Base64 Encoded Check Mark GIF

If you're having trouble downloading a check mark GIF, you can use this HTML file to generate one: 

<!DOCTYPE html>
<html>
<head>
    <title>Create Check Mark GIF</title>
    <style>
        body {
            display: flex;
            flex-direction: column;
            align-items: center;
            font-family: Arial, sans-serif;
        }
        canvas {
            border: 1px solid #ccc;
            margin: 20px;
        }
        button {
            padding: 10px 20px;
            background-color: #3085d6;
            color: white;
            border: none;
            border-radius: 4px;
            cursor: pointer;
        }
    </style>
</head>
<body>
    <h1>Simple Check Mark GIF Generator</h1>
    <canvas id="canvas" width="200" height="200"></canvas>
    <button id="downloadBtn">Download check-mark.gif</button>
    
    <script src="https://cdnjs.cloudflare.com/ajax/libs/gif.js/0.2.0/gif.js"></script>
    <script>
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');
        const downloadBtn = document.getElementById('downloadBtn');
        
        // Function to draw check mark
        function drawCheckMark(progress) {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Draw circle
            ctx.beginPath();
            ctx.arc(100, 100, 80, 0, 2 * Math.PI * progress);
            ctx.strokeStyle = '#4CAF50';
            ctx.lineWidth = 8;
            ctx.stroke();
            
            if (progress > 0.5) {
                // Draw check mark
                const checkProgress = (progress - 0.5) * 2; // 0 to 1
                
                ctx.beginPath();
                ctx.moveTo(70, 100);
                ctx.lineTo(70 + (30 * checkProgress), 100 + (30 * checkProgress));
                ctx.lineTo(70 + (30 * checkProgress) + (60 * checkProgress), 100 + (30 * checkProgress) - (60 * checkProgress));
                
                ctx.strokeStyle = '#4CAF50';
                ctx.lineWidth = 10;
                ctx.lineCap = 'round';
                ctx.lineJoin = 'round';
                ctx.stroke();
            }
        }
        
        // Create GIF
        downloadBtn.addEventListener('click', function() {
            const gif = new GIF({
                workers: 2,
                quality: 10,
                width: canvas.width,
                height: canvas.height
            });
            
            // Generate frames
            const frames = 30;
            for (let i = 0; i <= frames; i++) {
                drawCheckMark(i / frames);
                gif.addFrame(canvas, {copy: true, delay: 50});
            }
            
            gif.on('finished', function(blob) {
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'check-mark.gif';
                a.click();
            });
            
            gif.render();
        });
        
        // Initial drawing
        drawCheckMark(1);
    </script>
</body>
</html>
